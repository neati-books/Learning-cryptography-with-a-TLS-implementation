As we wished, All tutorials just worked fine without any adjusting. Please click buttons, hover mouse, and insert some text in upper examples. You will see the result. We can use libraries out of the box.

우리의 index.html을 vue.js 라이브러리를 사용하도록 고쳐보자.
위의 소스는 vue.js 튜토리얼에서 카피해 왔다. 우리가 바라듯이, 모든 튜토리얼이 어떤 변화나 편집이 없이 그냥 작동하였다. 한 번 버튼을 클릭해보고, 마우스를 문장 위에 움직여 보고, 텍스트를 필드에 입력해 보자. 그 결과를 볼 수 있을 것이다. 우리는 라이브러리를 적응 과정 없이 곧장 사용할 수 있다.

index.html

<html>
	<head>
		<script src="vue.js"></script>
		<script src="jquery.js"></script>
		<script src='index.js'></script>
	</head>
	<body>
		<div id="app">
			{{ message }}
		</div>
		<div id="app-2">
			<span v-bind:title="message">
				Hover your mouse over me for a few seconds
				to see my dynamically bound title!
			</span>
		</div>
		<div id="app-3">
			<span v-if="seen">Now you see me</span>
		</div>
		<div id="app-4">
			<ol>
				<li v-for="todo in todos">
					{{ todo.text }}
				</li>
			</ol>
		</div>
		<div id="app-5">
			<p>{{ message }}</p>
			<button v-on:click="reverseMessage">Reverse Message</button>
		</div>
		<div id="app-6">
			<p>{{ message }}</p>
			<input v-model="message">
		</div>
	</body>
</html>

index.js

$(function on_ready() {
	var app = new Vue({
		el: '#app',
		data: {
			message: 'Hello Vue!'
		}
	})
	var app2 = new Vue({
		el: '#app-2',
		data: {
			message: 'You loaded this page on ' + new Date().toLocaleString()
		}
	})
	var app3 = new Vue({
		el: '#app-3',
		data: {
			seen: true
		}
	})
	var app4 = new Vue({
		el: '#app-4',
		data: {
			todos: [
				{ text: 'Learn JavaScript' },
				{ text: 'Learn Vue' },
				{ text: 'Build something awesome' }
			]
		}
	})
	var app5 = new Vue({
		el: '#app-5',
		data: {
			message: 'Hello Vue.js!'
		},
		methods: {
			reverseMessage: function () {
				this.message = this.message.split('').reverse().join('')
			}
		}
	})
	var app6 = new Vue({
		el: '#app-6',
		data: {
			message: 'Hello Vue!'
		}
	})
});
